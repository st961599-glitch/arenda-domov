Оформление заявки

// Инициализация базы данных firebase.initializeApp(firebaseConfig); const db = firebase.firestore(); const tariffId = localStorage.getItem('selectedTariffId'); const tariffPrice = localStorage.getItem('selectedTariffPrice'); if (!tariffId) { window.location.href = 'index.html'; } // Выводим выбранный тариф на экран document.getElementById('tariff-summary').innerHTML = `

Выбранный период: ${tariffId}

Стоимость: ${Number(tariffPrice).toLocaleString('ru-RU')} ₽

`; // Отправка формы в базу данных клиентов document.getElementById('order-form').addEventListener('submit', async (e) => { e.preventDefault(); const name = document.getElementById('client-name').value; const phone = document.getElementById('client-phone').value; try { // Добавляем документ в коллекцию "Данные клиента" await db.collection("Данные клиента").add({ имя: name, телефон: phone, тариф: tariffId, цена: Number(tariffPrice), датаЗаявки: new Date().toISOString() }); alert('Спасибо! Заявка успешно отправлена.'); localStorage.clear(); window.location.href = 'index.html'; } catch (error) { console.error("Ошибка отправки:", error); alert('Ошибка отправки. Проверьте правила доступа Rules в Firebase.'); } });
